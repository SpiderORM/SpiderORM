name: Deploy Documentation to Netlify

on:
  push:
    branches:
      - main  # Ou o branch que voce deseja monitorar

jobs:
  build:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v3

    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.8'  # Especifique a versao do Python

    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install sphinx sphinx_rtd_theme

    - name: Build the documentation
      run: sphinx-build -b html . _build

    - name: Deploy to Netlify
      env:
        NETLIFY_AUTH_TOKEN: ${{ secrets.NETLIFY_AUTH_TOKEN }}  # Defina o token no GitHub Secrets
        NETLIFY_SITE_ID: ${{ secrets.NETLIFY_SITE_ID }}  # Defina o site ID no GitHub Secrets
      run: npx netlify deploy --dir=_build --prod --site=$NETLIFY_SITE_ID --auth=$NETLIFY_AUTH_TOKEN
